class ReponseCodeTests(TestCase):
    urls = 'tests.test_response'

    def test_should_throw_400_bad_fields(self):
        basic_model = BasicModel(text="test")
        basic_model.save()
        resp1 = self.client.put(
            '/html_new_model_viewset/{0}/'.format(basic_model.pk),
            '{"text": "bobby","bad": "bad"}',
            content_type='application/json')

        resp2 = self.client.patch(
            '/html_new_model_viewset/{0}/'.format(basic_model.pk),
            '{"bad": "bad"}',
            content_type='application/json')

        self.assertEqual(resp1.status_code, 400)
        self.assertEqual(resp2.status_code, 400)
The issue is: if you put/patch extra fields that aren't declared on your model serializer, it should throw a 400 or a 422, but right now it returns a 200 and fails silently. This seems like incorrect behavior.
