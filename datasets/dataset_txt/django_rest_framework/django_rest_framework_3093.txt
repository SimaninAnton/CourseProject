The following code should produce NameError: global name 'nonexistent_function' is not defined. Instead, d-r-f documentation renderer silences any exceptions and shows empty form with Content field. This makes debugging errors very hard.
from django import forms
from django.forms import fields

from djangorestframework.resources import FormResource
from djangorestframework.views import View
from djangorestframework.renderers import DocumentingHTMLRenderer, JSONRenderer
from djangorestframework.parsers import FormParser


class BrokenForm(forms.Form):
    testfield = fields.CharField()

    def __init__(self):
        nonexistent_function()

class BrokenResource(FormResource):
    form = BrokenForm

class BrokenView(View):
    parsers = (FormParser,)
    renderers = (JSONRenderer, DocumentingHTMLRenderer,)
    resource = BrokenResource

    def post(self, request, *args, **kwargs):
        return Response(status.HTTP_201_CREATED, {})
