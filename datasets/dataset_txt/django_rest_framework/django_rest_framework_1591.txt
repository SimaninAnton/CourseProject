from invoices import models
from rest_framework import serializers


class InvoiceItemSerializer(serializers.HyperlinkedModelSerializer):

    class Meta:
        model = models.InvoiceItem
        fields = (
            'id', 'url', 'invoice', 'description', 'price', 'discount'
        )
        extra_kwargs = {
            'url': {'view_name': 'invoice-item-detail'}
        }


class InvoiceSerializer(serializers.HyperlinkedModelSerializer):

    item_listing = serializers.HyperlinkedIdentityField(
        view_name='nested-invoice-item-list',
        read_only=True,
        lookup_field='id',
        lookup_url_kwarg='invoice_id'
    )

    class Meta:
        model = models.Invoice
        fields = (
            'id', 'url', 'time', 'customer', 'status', 'price', 'currency',
            'transaction_id', 'other_info', 'item_listing'
        )
