When doing a get request on the registered url in the view shown below, the self.request.method in get_serializer_class prints 'POST'.
When removing the get_serializer_class, the get method is called appropriately.
COLLECTION_TYPES = {
    'iati': CollectionSerializer,
}

class CollectionList(generics.GenericAPIView):
    """
    List and create collection
    """
    model = collection_models.Collection
    queryset = collection_models.Collection.objects.all()

    def get(self, request, *args, **kwargs):
        print('called')

        results = list()

        for collection in self.queryset:
            serializer = COLLECTION_TYPES[collection.type](collection)
            results.append(serializer.data)

        return Response(results)

    def post(self, request, *args, **kwargs):
        serializer = self.get_serializer_class(request)(request.data)
        serializer.is_valid(raise_exception=True)
        serializer.save()
        return Response(serializer.data, status=status.HTTP_201_CREATED)

    def get_serializer_class(self, request):
        print(self.request.method)
        if self.request.method in ('POST'):
            return COLLECTION_TYPES.get(self.request.data.type)
