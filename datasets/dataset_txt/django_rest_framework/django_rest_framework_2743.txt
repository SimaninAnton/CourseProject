The following proxy model behaves like django.contrib.auth.model.User but its custom manager excludes the user named 'admin'. But the related serializer does not exclude that user...
User = get_user_model()

class FilteredUserManager(models.Manager):
    def get_query_set(self):
        return super(FilteredUserManager,
                     self).get_query_set().exclude(username='admin')


class FilteredUser(User):
    objects = FilteredUserManager()

    class Meta:
        proxy = True

class UserSerializer(ModelSerializer):
    """A serializer for users data.
    """
    groups = SlugRelatedField(many=True, read_only=True, slug_field='name')

    class Meta:
        model = FilteredUser
        fields = ('id', 'username', 'first_name', 'email', 'last_name',
                  'last_login', 'date_joined', 'groups')
        ordering = ('username',)
