import json

from tornado.web import RequestHandler


class Method(RequestHandler):

    """
        Method function, for get/POST method got to data
        :param: method -- POST method data
                          GET method data
        :arg: get param name
    """

    def maigc_method(self, method, *args):

        while True:
            try:
                query = {}

                if method == 'post':    # POST Method
                    for param in args:
                        query[param] = json.dumps({k: self.get_argument(k) for k in self.request.arguments})[param]

                elif method == 'get':   # GET Method
                    for param in args:
                        query[param] = self.request.arguments[param]

                    return query

                else:
                    raise TypeError('Not support requested type')

            except KeyError or json.JSONEncoder as e:
                raise RuntimeError('Not found param:%s' % e)
