tornado can not work with multiple process mode on 4.2.1, raises RuntimeError with
"Cannot run in multiple processes: IOLoop instance has already been initialized. You cannot call IOLoop.instance() before calling start_processes()" message, or if it's change then documentation it's not updated
from tornado.web import Application
from tornado.web import RequestHandler
from tornado.ioloop import IOLoop
from tornado.httpserver import HTTPServer


class TestHandler(RequestHandler):
    def get(self):
          self.write("Ok")

class TestApplication(Application):

    def __init__(self):
        _handlers = [
            (r'/', TestHandler),
        ]

        settings = {
            "debug": True,
        }
        Application.__init__(self, _handlers, **settings)


if __name__ == '__main__':
    server = HTTPServer(TestApplication())
    server.bind(8002)
    server.start(0)
    IOLoop.current().start()
