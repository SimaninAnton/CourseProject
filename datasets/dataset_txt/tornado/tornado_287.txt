Now, We have a tornado web application need to be deployed by using uwsgi, I want to share my personal experience for other guys.
I write a tornado web app named warriors, below is my python app code file which named
start.py
 import tornado.web
 import tornado.wsgi

 class MainHandler(tornado.web.RequestHandler):
     def get(self):
         self.write("Hello, world")

 tornado_app = tornado.web.Application([
     (r"/", MainHandler),
 ])
 application = tornado.wsgi.WSGIAdapter(tornado_app)
the file stored in path '/root/warrior' and python virtualenv path is '/root/py27'
below is my uwsgi config file named
uwsgi.ini
[uwsgi]
processes = 4

socket = /etc/uwsgi/warriors.sock
chmod-socket = 666

pidfile = /etc/uwsgi/warriors.pid

chdir = /root/warrior
wsgi-file = start.py

daemonize = /var/logs/uwsgi/warriors.log
log-maxsize = 20000000

disable-logging = true
vacuum = true
virtualenv = /root/py27
nginx file
server {
    listen  80;
    server_name default_server;
    access_log /var/logs/nginx/warriors.access.log main buffer=32k flush=5s;
    error_log /var/logs/nginx/warriors.error.log;
    client_max_body_size 200M;

    location ~ / {
        include uwsgi_params;
        uwsgi_pass unix:////etc/uwsgi/warriors.sock;
    }
}
