class SelectThreshold(_AbstractUnivariateFilter):
    """Filter: Select based on threshold of p_values

    Parameters
    ==========
    score_func: callable
        function taking two arrays X and y, and returning 2 arrays:
        both scores and pvalues

    threshold: float, optional
        threshold above which to keep
    """

    def __init__(self, score_func=f_classif, threshold=0.05):
        self.threshold = threshold
        _AbstractUnivariateFilter.__init__(self, score_func)

    def _get_support_mask(self):
        return self.pvalues_ >= self.threshold
