Expected Behavior
In flask 1.0, calling flask.abort(400) returns a 500 error in debug mode
import flask
import werkzeug

APP = flask.Flask(__name__)
print(flask.__version__)
print(werkzeug.__version__)

@APP.route('/<arg>')
def index(arg):
    return flask.abort(400)

APP.debug = True
APP.run()
