#   send 410 page gone response for non-existent pages to avoid cluttering error logs
    RewriteMap movedPageMap txt:conf/movedPage.map
    RewriteCond %{DOCUMENT_ROOT}%{REQUEST_URI} !-f
    RewriteCond %{DOCUMENT_ROOT}%{REQUEST_URI} !-d
    RewriteCond ${movedPageMap:$1}  >""
    RewriteRule ^.*$ /movedPage.php [L]
