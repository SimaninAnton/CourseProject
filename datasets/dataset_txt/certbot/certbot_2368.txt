Now that Boulder supports Must Staple and ECDSA, it would be nice to have test cases for them. We would have to manually generate CSRs for them, like so:
Must Staple:
openssl req -new -sha256 -key rsa.key -subj "/" -reqexts SAN -config <(cat /etc/ssl/openssl.cnf <(printf "[SAN]\nsubjectAltName=DNS:must-staple.le.wtf\n1.3.6.1.5.5.7.1.24=DER:30:03:02:01:05")) -out must-staple.csr.pem
ECDSA:
openssl ecparam -genkey -name secp384r1 > privkey-p384.pem ; openssl req -new -sha256 -key privkey-p384.pem -subj "/" -reqexts SAN -config <(cat /etc/ssl/openssl.cnf <(printf "[SAN]\nsubjectAltName=DNS:ecdsa.le.wtf")) -outform der -out csr-p384.der
