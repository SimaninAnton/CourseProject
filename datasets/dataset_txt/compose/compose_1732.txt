This is due to it trying to parse the timeout and interval parameters more than once.
Here are example yml files.
base.yml
services:
  demo:
    image: foobar:latest
    entrypoint: ["/bin/bash", "-c", "echo foobar"]
    healthcheck:
      test: ["CMD", "/health.sh"]
      interval: 10s
      timeout: 5s
      retries: 36
extends.yml
services:
  demo:
    extends:
      file: base.yml
      service: demo
