import os
import sys

cur_dir = os.getcwd()
# print(cur_dir)
python_path = os.environ.get('PYTHONPATH', '')
if cur_dir not in python_path:
    if python_path:
        python_path = python_path + ':' + cur_dir
    else:
        python_path = cur_dir
os.environ['PYTHONPATH'] = python_path
