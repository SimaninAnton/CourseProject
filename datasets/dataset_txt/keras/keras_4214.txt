def make_ixm_lstm(n_frame=100,
                  n_freq=276,
                  n_layers=4,
                  n_units=300,
                  n_bins=10,
                  loss='mean_squared_error',
                  eps=0.1,
                  verbose=False):
    inputs = Input(shape=(n_frame, n_freq))
    lstm = LSTM(n_units, return_sequences=True)(inputs)
    for _ in range(n_layers - 1):
        lstm = LSTM(n_units, return_sequences=True)(lstm)
    lstm_out = LSTM(n_bins, return_sequences=True)(lstm)
    sigmoid_out = Activation('sigmoid')(lstm_out)
    model = Model(input=inputs, output=sigmoid_out)
    model.compile(optimizer=Adam(), loss=loss, metrics=['binary_accuracy',],)
    print 'DONE MAKING LSTM MODEL'
    return model
