K.batch_dot appears to work fine with Tensorflow backend but gives garbage with Theano; bd[0] in the script below isn't even symmetric.
from keras import backend as K
import numpy as np
from numpy.testing import assert_allclose

batch = K.variable(np.random.randn(50, 8, 5))

bd = K.eval(K.batch_dot(batch, K.permute_dimensions(batch, [0, 2, 1])))

d = K.eval(K.dot(batch[0], K.permute_dimensions(batch[0], [1, 0])))

assert_allclose(bd[0], d, atol=1e-05)
