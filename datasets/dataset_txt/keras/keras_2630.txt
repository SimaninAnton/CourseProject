def SequenceAE():
    x = Input(shape=(maxlen,embedding_dims))
    tied = GRU(embedding_dims)
    d1 = tied(x)
    z = Dense(50,activation='relu')(d1)
    d2 = RepeatVector(maxlen)(z)
    d3 = LSTM(embedding_dims,return_sequences=True)(d2)
    
    out = Dense(1, activation='sigmoid', name='classification')(z)
    
    train_model = Model(input=[x], output=[d3])
    train_model.compile(optimizer='rmsprop',
                  loss = 'binary_crossentropy')
    test_model = Model(input=x, output=out)
    test_model.compile(optimizer='adam',
                  loss = 'binary_crossentropy',metrics=['accuracy'])
    return train_model, test_model
