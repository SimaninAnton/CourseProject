def crop_one_img(img, label, strip=256):
    samples = []
    sample_labels = []
    loc_0 = 0
    while loc_0 < label.shape[0]:
        loc_1 = 0
        if loc_0 + strip < label.shape[0]:
            while loc_1 < label.shape[1]:
                if loc_1 + strip < label.shape[1]:
                    samples.append(img[loc_0: loc_0 + strip, loc_1: loc_1 + strip, :])
                    sample_labels.append(label[loc_0: loc_0 + strip, loc_1: loc_1 + strip])
                else:
                    samples.append(img[loc_0: loc_0 + strip, -strip:, :])
                    sample_labels.append(label[loc_0: loc_0 + strip, -strip:])
                loc_1 += strip

        else:
            while loc_1 < label.shape[1]:
                if loc_1 + strip < label.shape[1]:
                    samples.append(img[-strip:, loc_1: loc_1 + strip, :])
                    sample_labels.append(label[-strip:, loc_1: loc_1 + strip])
                else:
                    samples.append(img[-strip:, -strip:, :])
                    sample_labels.append(label[-strip:, -strip:])
                loc_1 += strip

        loc_0 += strip

    return samples, sample_labels


def get_img(imgnames):
    SAMPLES = []
    SAMPLE_LABELS = []
    for imgname in imgnames:
        img = np.asarray(Image.open('./data/all/%s/img.png' % (imgname)), dtype='float16')  # 默认64位
        lbl = np.asarray(Image.open('./data/all/%s/label.png' % (imgname)))
        if img.shape[0] > img.shape[1]:
            img = np.transpose(img, (1, 0, 2))
            lbl = np.transpose(lbl)
        img = img / 255.0

        with open('./data/all/%s/label_names.txt' % imgname) as f:
            names = f.readlines()
            names = [i.strip() for i in names]

        label = np.zeros((lbl.shape[0], lbl.shape[1]))
        for na, la in zip(names, range(len(names))):
            if la > 0:
                label[lbl == la] = labelme_dic[na]

        samples, sample_labels = crop_one_img(img, label)
        SAMPLES.extend(samples)
        SAMPLE_LABELS.extend(sample_labels)

    return np.asarray(SAMPLES, dtype='float16'), to_categorical(np.asarray(SAMPLE_LABELS, dtype='float16'))


def get_train_batch(train_data_names, batch_size=5):
    while 1:
        for i in range(0, len(train_data_names), batch_size):
            X_train, Y_train = get_img(train_data_names[i:i+batch_size])
            yield (X_train, Y_train)
