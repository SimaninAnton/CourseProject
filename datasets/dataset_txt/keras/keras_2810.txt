The code below causes ValueError: Optimizer weight shape (5,) not compatible with provided weight shape (1,).
#!/usr/bin/env python

import os
import numpy as np
from keras.models import Sequential, load_model
from keras.layers import InputLayer, Dense


inner = Sequential([
    InputLayer([3]),
    Dense(5)
])

model = Sequential([
    InputLayer([3]),
    inner,
    Dense(1)
])

inner.trainable = False
model.compile('adam', 'mse')

# to set optimizer's weights
model.train_on_batch(np.random.uniform(-1, 1, [100, 3]), np.random.uniform(-1, -1, [100, 1]))

model.save("./hoge.h5")
try:
    load_model("./hoge.h5")
except ValueError as e:
    print(e)
finally:
    os.remove("./hoge.h5")
