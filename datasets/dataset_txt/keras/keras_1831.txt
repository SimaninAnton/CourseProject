The result of K.mean applied onto a Keras tensor is not a Keras tensor anymore.
import keras

a = keras.Input(shape=(20, 20))
b = keras.backend.mean(a)

print(keras.backend.is_keras_tensor(a))  # Is true
print(keras.backend.is_keras_tensor(b))  # Is false
