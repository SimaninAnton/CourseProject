I want to get links from a web page and follow them to parse table with statistics, but somehow I don't see that any links would be grabbed and followed to web page that has data. Here is my script:
from scrapy.linkextractors import LinkExtractor
from scrapy.spiders import CrawlSpider, Rule
from scrapy.item import Item, Field

class CNNVDItem(Item):
    title = Field()

class CNNVDSpider(CrawlSpider):
    name = 'cnnvd'
    allowed_domains = 'www.cnnvd.org.cn'
    start_urls = [
        'http://www.cnnvd.org.cn/vulnerability/index/p/1/',
    ]

    rules = (
        Rule(LinkExtractor(allow=r'\?&amp;p=\d{1,5}'), follow=True, ),
        Rule(LinkExtractor(allow=r'/vulnerability/show/cv_id/(\d+)'), callback='parse_detail', follow=True, ),
    )

    def parse_detail(self, response):
        item = CNNVDItem()
        title = response.xpath('//table[@class="details"]/tr[1]/td[2]/text()').extract()[0]
        item['title'] = title
        yield item
