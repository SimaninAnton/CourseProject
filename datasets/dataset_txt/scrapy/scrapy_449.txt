import scrapy
class FarmersSpider(scrapy.Spider):
name = 'farmers'
allowed_domains = ['www.farmerscompress.com']
login_url = 'https://www.farmerscompress.com/ProcessUser.aspx'
start_urls = [login_url]
def parse(self, response):
 yield scrapy.FormRequest(url=self.login_url,
  formdata={'T1': 'user', 'T2': 'pass', 'B2': 'Login'},
  callback=self.after_login)

def after_login(self, response):
 tes = response.css('#Label1').extract()
 yield {'tes': 'tes', 'tes': tes}
